% when "sigaction" is used as a type, to make it bold (not putting to index) use @= trick
% and add @:sigaction}\&{sigaction@> to put it to index (if \noinx is not used)
\def\beginbold{\egroup\bgroup\let\vb=\&\bgroup}
\def\endbold{\egroup\egroup\bgroup}

% use @= trick for inner variables of structures from header files not to put them in index
% (in \noinx is not used)
\def\beginhide{\egroup\bgroup\let\vb=\\\bgroup}
\def\endhide{\egroup\egroup\bgroup}
